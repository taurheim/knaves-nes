50 {
51  // Take care of pending interrupts
52  if (interrupts.size() > 0)
53  {
54  bool interruptShouldExecute = true;
55 
56  // Ignore interrupt for now if disabled flag is set, besides NMI, which always get executed
57  if (interrupts.front() != Interrupt::Nmi && hasStatusFlag(STATUS_INTERRUPT))
58  {
59  interruptShouldExecute = false;
60  }
61 
62  if (interruptShouldExecute)
63  {
64  reg_pc = executeInterrupt(interrupts.front());
65  interrupts.pop_front();
66 
67  // Interrupt takes 7 cycles to complete
68  return INTERRUPT_CYCLES;
69  }
70  }
71 
72  // Reset internal flags
73  branchTaken = false;
74  pageBoundaryCrossed = false;
75 
76  // Fetch and decode opcode
77  opcode = read(reg_pc);
78  opcodeHistory <<= 8;
79  opcodeHistory |= opcode & 0xFF;
80  setStatusFlag(STATUS_EMPTY);
81  it = opcode_table.find(opcode);
82 
83  // Invalid opcode
84  if (it == opcode_table.end())
85  {
86  throw InvalidOpcodeException(opcode);
87  }
88 
89  // Execute opcode
90  entry = it->second;
91 
92  // Log instruction?
93  if (Config::instance().doInstructionLogging)
94  {
95  log(&entry);
96  }
97 
98  src = (this->*entry.addressPtr)(entry.dummy);
99  (this->*entry.functionPtr)();
100 
101 
102  // Create opcode_entry for PPU
103  //opcode_entry ppuOpcodeEntry = entry;
104  unsigned short cycles = entry.cycles;
105 
106  // Add extra cycles if branch was taken
107  if (branchTaken)
108  {
109  if (pageBoundaryCrossed)
110  {
111  // TODO: shouldn't this be +=2? but that fails instr_misc/03-dummy_reads
112  cycles += 2;
113  }
114  else
115  {
116  cycles++;
117  }
118  }
119  // Add an extra cycle if crossing page boundary
120  else if (entry.cycles_extra && pageBoundaryCrossed)
121  {
122  cycles++;
123  }
124

799 unsigned short cpu::executeInterrupt(const enum Interrupt &interrupt)
800 {
801  switch (interrupt)
802  {
803  case Interrupt::Nmi:
804  pushStack((reg_pc >> 8) & 0xFF);
805  pushStack(reg_pc & 0xFF);
806  pushStack(reg_status & ~STATUS_BRK);
807  setStatusFlag(STATUS_INTERRUPT);
808  return (read(INTERRUPT_NMI_HIGH) << 8) | read(INTERRUPT_NMI_LOW);
809  break;
810 
811  case Interrupt::Irq:

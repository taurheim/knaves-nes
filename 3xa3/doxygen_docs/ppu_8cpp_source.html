462  unsigned char patternPlane1, patternPlane2, paletteUpperBits;
463  unsigned char tileX, tileY, tileScrollY, tileScrollX;
464 
465  // Select base address for pattern table
466  if (ppu_control & PPU_CONTROL_BG_PATTERN_ADDR)
467  {
468  patternTableAddr = 0x1000;
469  }
470  else
471  {
472  patternTableAddr = 0x0000;
473  }
474 
475  // Iterate through tiles in name table
476  for (int tileNum = 0; tileNum < NAME_TABLE_WIDTH; tileNum++)
477  {
478  switch ((vram_address >> 10) & 0x03)
479  {
480  case 0: nameTableAddr = 0x2000; break;
481  case 1: nameTableAddr = 0x2400; break;
482  case 2: nameTableAddr = 0x2800; break;
483  case 3: nameTableAddr = 0x2C00; break;
484  }
485 
486  attributeTableAddr = nameTableAddr + 0x03C0;
487  tileX = vram_address & 0x1F;
488  tileY = (vram_address >> 5) & 0x1F;
489  tileScrollY = (vram_address >> 12) & 0x07;
490  tileScrollX = scroll_x;
491  tileAddress = nameTableAddr | (vram_address & 0x03FF);
492 
493  // Iterate through pixels in tile
494  for (int pixelNum = 0; pixelNum < TILE_WIDTH; pixelNum++)
495  {
496  // Get pattern data
497  tileIndex = read(tileAddress);
498  patternPlane1 = read(patternTableAddr + (tileIndex << 4) + tileScrollY);
499  patternPlane2 = read(patternTableAddr + (tileIndex << 4) + tileScrollY + 8);
500 
501  // Get attribute data
502  //attributeAddress = attributeTableAddr | (vram_address & 0x0C00) | ((vram_address >> 4) & 0x38) | ((vram_address >> 2) & 0x07);
503  attributeAddress = attributeTableAddr | ( ((((tileY * TILE_WIDTH) + tileScrollY) / 32) * (SCREEN_WIDTH / 32)) + (((tileX * TILE_WIDTH) + tileScrollX) / 32) );
504  attributeValue = read(attributeAddress);
505  groupIndex = (((tileX % 4) & 0x2) >> 1) + ((tileY % 4) & 0x2);
506  paletteUpperBits = ((attributeValue >> (groupIndex<<1)) & 0x3) << 2;
507 
508  // Get color bits
509  paletteIndex = paletteUpperBits;
510  paletteIndex |= patternPlane1 & (0x80 >> tileScrollX) ? 0x1 : 0;
511  paletteIndex |= patternPlane2 & (0x80 >> tileScrollX) ? 0x2 : 0;
512 
513  // If color bits from pattern tables are zero => transparent pixel
514  if ((paletteIndex & 0x3) == 0)
515  {
516  _renderer->setTransparentPixel((tileNum << 3) + pixelNum, scanline);
517  }
518  else

520  // Render pixel to buffer
521  _renderer->setPixel(PixelType::BackgroundTile, (tileNum << 3) + pixelNum, scanline, palette_table[read(ADDR_PALETTE_BG + paletteIndex)]);
522  }
523

650 {

652  address = mirrorNameTables(address);
653

667  if (address == 0x3F10 ||
